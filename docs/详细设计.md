### 1.2 EmbeddingGenerator (向量生成)

#### 职责
- 将文本转换为向量表示
- 支持多种嵌入模型
- 本地/远程模型适配

#### 接口

```typescript
export class EmbeddingGenerator {
  private model: any = null;
  private cache: Map<string, number[]> = new Map();
  private vectorDim: number = 384;
  
  async initialize(): Promise<void> {
    // 尝试加载本地模型
    try {
      // 使用轻量级模型
      this.model = await pipeline('feature-extraction', 'Xenova/all-MiniLM-L6-v2');
      this.vectorDim = 384;
    } catch (e) {
      console.warn('本地模型加载失败，使用备用方案');
    }
  }
  
  async embed(text: string): Promise<number[]> {
    // 1. 检查缓存
    if (this.cache.has(text)) {
      return this.cache.get(text)!;
    }
    
    // 2. 生成向量
    let vector: number[];
    if (this.model) {
      vector = await this.model(text, {
        pooling: 'mean',
        normalize: true
      });
      vector = Array.from(vector.data);
    } else {
      // 备用: 关键词向量化
      vector = this.keywordEmbedding(text);
    }
    
    // 3. 缓存
    this.cache.set(text, vector);
    return vector;
  }
  
  private keywordEmbedding(text: string): number[] {
    // 简化的关键词向量化
    const vector = new Array(this.vectorDim).fill(0);
    const words = text.toLowerCase().split(/\s+/);
    
    let idx = 0;
    for (const word of words) {
      for (let i = 0; i < word.length && idx < this.vectorDim; i++) {
        vector[idx] += word.charCodeAt(i);
        idx = (idx + 1) % this.vectorDim;
      }
    }
    
    // 归一化
    const norm = Math.sqrt(vector.reduce((sum, v) => sum + v * v, 0));
    return vector.map(v => v / (norm || 1));
  }
  
  cosineSimilarity(a: number[], b: number[]): number {
    let dotProduct = 0;
    let normA = 0;
    let normB = 0;
    
    for (let i = 0; i < a.length; i++) {
      dotProduct += a[i] * b[i];
      normA += a[i] * a[i];
      normB += b[i] * b[i];
    }
    
    return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB) || 1);
  }
}
```

### 1.3 MemorySystem (记忆系统核心)

#### 职责
- 记忆的 CRUD 操作
- 向量搜索
- 分类分析

#### 接口

```typescript
export interface Memory {
  id: string;
  taskDescription: string;
  category: string;
  subCategory: string;
  cluster: string;
  content: {
    task: string;
    process: string;
    result: string;
    insights: string[];
  };
  embedding: number[];
  createdAt: Date;
  updatedAt: Date;
  lastAccessedAt: Date;
  version: number;
  diffs: any[];
  mergedFrom: string[];
  confidence: number;
  usageCount: number;
  importance: 'critical' | 'high' | 'medium' | 'low';
  tags: string[];
  encrypted: boolean;
  encryptedContent?: string;
  autoCleanup?: boolean;
  cleanupAt?: Date;
  summary?: string;
}

export interface MemoryInput {
  content: {
    task: string;
    process: string;
    result: string;
    insights: string[];
  };
  category: string;
  subCategory: string;
  importance: 'critical' | 'high' | 'medium' | 'low';
  tags?: string[];
}

export class MemorySystem {
  private config: MemoryConfig;
  private memories: Map<string, Memory> = new Map();
  private categoryIndex: Map<string, Set<string>> = new Map();
  private embeddingGenerator: EmbeddingGenerator;
  private vectorStore: VectorStore;
  private dataDir: string;
  
  constructor(dataDir: string, config: MemoryConfig) {
    this.dataDir = dataDir;
    this.config = config;
    this.embeddingGenerator = new EmbeddingGenerator();
    this.vectorStore = new LocalVectorStore(dataDir);
  }
  
  async initialize(): Promise<void> {
    // 确保目录存在
    if (!fs.existsSync(this.dataDir)) {
      fs.mkdirSync(this.dataDir, { recursive: true });
    }
    
    // 初始化向量生成器
    await this.embeddingGenerator.initialize();
    
    // 初始化向量存储
    await this.vectorStore.initialize();
    
    // 加载已有记忆
    await this.loadMemories();
  }
  
  // 搜索记忆
  async search(query: string, options: {
    limit: number;
    threshold: number;
  }): Promise<Memory[]> {
    // 1. 生成查询向量
    const queryVector = await this.embeddingGenerator.embed(query);
    
    // 2. 向量搜索
    const candidates = await this.vectorStore.search(queryVector, options.limit * 3);
    
    // 3. 计算相似度并过滤
    const results: { memory: Memory; score: number }[] = [];
    
    for (const candidate of candidates) {
      const memory = this.memories.get(candidate.id);
      if (!memory) continue;
      
      const score = this.embeddingGenerator.cosineSimilarity(
        queryVector,
        memory.embedding
      );
      
      if (score >= options.threshold) {
        results.push({ memory, score });
      }
    }
    
    // 4. 排序返回
    results.sort((a, b) => b.score - a.score);
    
    // 5. 更新访问时间 (强化)
    for (const r of results.slice(0, options.limit)) {
      this.enhance(r.memory.id);
    }
    
    return results.slice(0, options.limit).map(r => r.memory);
  }
  
  // 存储记忆
  async store(input: MemoryInput): Promise<Memory> {
    const id = this.generateMemoryId(input.category, input.subCategory);
    
    // 生成向量
    const text = `${input.content.task} ${input.content.process}`;
    const embedding = await this.embeddingGenerator.embed(text);
    
    const memory: Memory = {
      id,
      taskDescription: input.content.task,
      category: input.category,
      subCategory: input.subCategory,
      cluster: `${input.subCategory}-cluster`,
      content: input.content,
      embedding,
      createdAt: new Date(),
      updatedAt: new Date(),
      lastAccessedAt: new Date(),
      version: 1,
      diffs: [],
      mergedFrom: [],
      confidence: 0.8,
      usageCount: 0,
      importance: input.importance,
      tags: input.tags || [],
      encrypted: false,
      autoCleanup: false
    };
    
    // 保存
    this.memories.set(id, memory);
    this.addToCategoryIndex(memory);
    await this.saveMemory(memory);
    await this.vectorStore.add(id, embedding, { 
      text: text.substring(0, 1000) 
    });
    
    return memory;
  }
  
  // 强化记忆
  async enhance(memoryId: string): Promise<void> {
    const memory = this.memories.get(memoryId);
    if (!memory) return;
    
    memory.usageCount++;
    memory.lastAccessedAt = new Date();
    memory.updatedAt = new Date();
    
    // 提升重要性
    if (memory.importance !== 'critical') {
      const levels: ('low' | 'medium' | 'high' | 'critical')[] = 
        ['low', 'medium', 'high', 'critical'];
      const idx = levels.indexOf(memory.importance);
      if (idx < levels.length - 1 && memory.usageCount > 3) {
        memory.importance = levels[idx + 1];
      }
    }
    
    await this.saveMemory(memory);
  }
  
  // 获取所有记忆
  getAllMemories(): Memory[] {
    return Array.from(this.memories.values());
  }
  
  // 获取统计
  getStats(): {
    total: number;
    byCategory: Record<string, number>;
    byImportance: Record<string, number>;
    averageUsage: number;
  } {
    const memories = this.getAllMemories();
    
    const byCategory: Record<string, number> = {};
    const byImportance: Record<string, number> = {};
    let totalUsage = 0;
    
    for (const m of memories) {
      byCategory[m.category] = (byCategory[m.category] || 0) + 1;
      byImportance[m.importance] = (byImportance[m.importance] || 0) + 1;
      totalUsage += m.usageCount;
    }
    
    return {
      total: memories.length,
      byCategory,
      byImportance,
      averageUsage: memories.length > 0 ? totalUsage / memories.length : 0
    };
  }
  
  // 私有方法
  private async loadMemories(): Promise<void> {
    if (!fs.existsSync(this.dataDir)) return;
    
    const categories = fs.readdirSync(this.dataDir);
    
    for (const category of categories) {
      const categoryPath = path.join(this.dataDir, category);
      if (!fs.statSync(categoryPath).isDirectory()) continue;
      
      const subCategories = fs.readdirSync(categoryPath);
      
      for (const subCategory of subCategories) {
        const subCategoryPath = path.join(categoryPath, subCategory);
        if (!fs.statSync(subCategoryPath).isDirectory()) continue;
        
        const files = fs.readdirSync(subCategoryPath);
        
        for (const file of files) {
          if (!file.endsWith('.json') || file === 'index.json') continue;
          
          try {
            const content = fs.readFileSync(
              path.join(subCategoryPath, file), 
              'utf-8'
            );
            const memory = JSON.parse(content);
            memory.createdAt = new Date(memory.createdAt);
            memory.updatedAt = new Date(memory.updatedAt);
            memory.lastAccessedAt = new Date(memory.lastAccessedAt);
            
            this.memories.set(memory.id, memory);
            this.addToCategoryIndex(memory);
          } catch (e) {
            console.error(`加载记忆失败: ${file}`, e);
          }
        }
      }
    }
    
    // 重建向量索引
    if (this.memories.size > 0) {
      const memoriesToIndex = Array.from(this.memories.values()).map(m => ({
        id: m.id,
        text: `${m.content.task} ${m.content.process}`
      }));
      await this.vectorStore.rebuild(memoriesToIndex);
    }
  }
  
  private async saveMemory(memory: Memory): Promise<void> {
    const dir = path.join(
      this.dataDir, 
      memory.category, 
      memory.subCategory
    );
    
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }
    
    const filePath = path.join(dir, `${memory.id}.json`);
    await fs.promises.writeFile(filePath, JSON.stringify(memory, null, 2));
  }
  
  private addToCategoryIndex(memory: Memory): void {
    if (!this.categoryIndex.has(memory.category)) {
      this.categoryIndex.set(memory.category, new Set());
    }
    this.categoryIndex.get(memory.category)!.add(memory.id);
  }
  
  private generateMemoryId(category: string, subCategory: string): string {
    const now = new Date();
    const date = now.toISOString().slice(0, 10).replace(/-/g, '');
    const time = now.toTimeString().slice(0, 5).replace(/:/g, '');
    const uuid = require('uuid').v4().slice(0, 6);
    return `${category.slice(0, 2)}-${subCategory.slice(0, 2)}-${date}-${time}-${uuid}`;
  }
}
```

```typescript
export interface VectorStore {
  initialize(): Promise<void>;
  add(id: string, vector: number[], metadata?: any): Promise<void>;

  delete(id: string): Promise<void>;

// 本地向量存储实现
export class LocalVectorStore implements VectorStore {
  private vectors: Map<string, number[]> = new Map();
  private metadata: Map<string, any> = new Map();
  private indexPath: string;
  
  constructor(dataDir: string) {
    this.indexPath = path.join(dataDir, 'vector-index.json');
  }
  
  async initialize(): Promise<void> {
    if (fs.existsSync(this.indexPath)) {
      try {
        const data = JSON.parse(fs.readFileSync(this.indexPath, 'utf-8'));
        this.vectors = new Map(Object.entries(data.vectors || {}));
        this.metadata = new Map(Object.entries(data.metadata || {}));
      } catch (e) {
        console.warn('向量索引加载失败');
      }
    }
  }
  
  async add(id: string, vector: number[], metadata?: any): Promise<void> {
    this.vectors.set(id, vector);
    if (metadata) {
      this.metadata.set(id, metadata);
    }
  }
  
  async search(vector: number[], limit: number): Promise<SearchResult[]> {
    const results: SearchResult[] = [];
    
    for (const [id, storedVector] of this.vectors.entries()) {
      // 简单余弦相似度
      let dotProduct = 0;
      let normA = 0;
      let normB = 0;
      
      for (let i = 0; i < vector.length; i++) {
        dotProduct += vector[i] * storedVector[i];
        normA += vector[i] * vector[i];
        normB += storedVector[i] * storedVector[i];
      }
      
      const score = dotProduct / (Math.sqrt(normA) * Math.sqrt(normB) || 1);
      
      results.push({
        id,
        score,
        metadata: this.metadata.get(id)
      });
    }
    
    // 排序并返回 top N
    results.sort((a, b) => b.score - a.score);
    return results.slice(0, limit);

    this.metadata.clear();

  }
```

```typescript
export class TriggerAnalyzer {
  // 关键信息
  private criticalPatterns = [
    /记住|别忘了|提醒我|提醒/i,
    /密码|账号|身份|银行卡|信用卡/i,
    /\+\d{10,}/,                    // 电话

    /password|account|identity/i
  ];

    /决定|选择|采用|否决|定下来/i,
    /配置|安装|部署|设置/i,
    /prefer|decide|choose|use|adopt/i,
    /config|install|deploy|setup/i
  ];
  
  // 一般工作
  private mediumPatterns = [
    /实现|修复|创建|帮助|完成|解决了/i,
    /写|做|生成|处理|分析/i,
    /implement|fix|create|help|complete|build/i,
    /write|make|generate|process|analyze/i
  ];
  
  // 低价值
  private lowPatterns = [
    /^你好|^谢谢|^再见|^好的/i,
    /hi|hello|thanks|bye|ok|okay/i,
    /^请问|^没什么/i
  ];
  
  analyze(content: string): 'critical' | 'high' | 'medium' | 'low' {
    const lower = content.toLowerCase();
    
    if (this.criticalPatterns.some(p => p.test(lower))) {
      return 'critical';
    }
    if (this.highPatterns.some(p => p.test(lower))) {
      return 'high';
    }
    if (this.lowPatterns.some(p => p.test(lower))) {
      return 'low';
    }
    if (this.mediumPatterns.some(p => p.test(lower))) {
      return 'medium';
    }
    
    // 默认中等
    return 'medium';
  }
  
  extractTags(content: string): string[] {
    const tags: string[] = [];
    const lower = content.toLowerCase();
    
    // 标签映射
    const tagPatterns: Record<string, RegExp[]> = {
      '代码': [/代码|编程|函数|脚本/i, /code|program|script/i],
      '配置': [/配置|设置|安装/i, /config|setup|install/i],
      '数据': [/数据|分析|统计/i, /data|analysis|stat/i],
      '文档': [/文档|报告|文章/i, /doc|report|article/i],
      '决策': [/决定|选择|采用/i, /decide|choose|adopt/i],
      '问题': [/错误|bug|问题|故障/i, /error|bug|issue|problem/i],
      '完成': [/完成|解决了|实现了/i, /done|completed|implemented/i]
    };
    
    for (const [tag, patterns] of Object.entries(tagPatterns)) {
      if (patterns.some(p => p.test(lower))) {
        tags.push(tag);
      }
    }
    
    return tags;
  }
}
```

```typescript
export class CategoryAnalyzer {
  private categoryKeywords: Record<string, string[]> = {
    '代码开发': ['代码', '编程', '脚本', 'Python', 'JavaScript', '开发', '函数', 'bug'],
    '系统运维': ['部署', '服务器', '运维', 'Docker', '配置', '安装', '备份'],
    '文档撰写': ['文档', '报告', '文章', '写作', '撰写', '总结'],
    '数据分析': ['分析', '数据', '统计', '图表', '可视化', 'CSV', 'Excel'],
    '创作设定': ['角色', '故事', '世界观', '创作', '小说', '设定'],
    '用户偏好': ['喜欢', '偏好', '习惯', '讨厌', '想要', '口味'],
    '日常事务': ['日程', '邮件', '安排', '提醒', '天气', '时间']
  };
  
  analyze(content: string): { category: string; subCategory: string } {
    const lower = content.toLowerCase();
    let bestCategory = '其他';
    let maxScore = 0;
    
    for (const [category, keywords] of Object.entries(this.categoryKeywords)) {
      let score = 0;
      for (const keyword of keywords) {
        if (lower.includes(keyword.toLowerCase())) {
          score += 1;
        }
      }
      if (score > maxScore) {
        maxScore = score;
        bestCategory = category;
      }
    }
    
    return {
      category: bestCategory,
      subCategory: this.getSubCategory(content, bestCategory)
    };
  }
  
  private getSubCategory(content: string, category: string): string {
    // 根据内容进一步细分
    const lower = content.toLowerCase();
    
    if (category === '代码开发') {
      if (/python|脚本/.test(lower)) return '脚本';
      if (/web|前端|html|css|react/.test(lower)) return '前端';
      if (/api|后端|server/.test(lower)) return '后端';
      return '通用';
    }
    
    if (category === '系统运维') {
      if (/docker|容器/.test(lower)) return '容器';
      if (/部署|发布/.test(lower)) return '部署';
      if (/备份|恢复/.test(lower)) return '备份';
      return '通用';
    }
    
    return '通用';
  }
}
```

### 1.7 AutoRecall (自动召回)

#### 职责
- 在 AI 回复前搜索相关记忆
- 格式化并注入到上下文

```typescript
export class AutoRecall {
  private memorySystem: MemorySystem;
  private config: MemoryConfig;
  
  constructor(memorySystem: MemorySystem, config: MemoryConfig) {
    this.memorySystem = memorySystem;
    this.config = config;
  }
  
  async execute(event: any): Promise<{ prependContext: string } | void> {
    // 1. 提取用户输入
    const prompt = this.extractUserPrompt(event.prompt);
    if (!prompt || prompt.length < 3) return;
    
    try {
      // 2. 搜索相关记忆
      const memories = await this.memorySystem.search(prompt, {
        limit: this.config.recallLimit,
        threshold: this.config.recallThreshold
      });
      
      if (memories.length === 0) return;
      
      // 3. 格式化
      const context = this.formatMemories(memories);
      
      // 4. 返回注入
      return {
        prependContext: context
      };
    } catch (e) {
      console.error('自动召回失败:', e);
    }
  }
  
  private extractUserPrompt(prompt: any): string {
    if (typeof prompt === 'string') {
      return prompt;
    }
    if (Array.isArray(prompt)) {
      // 取最后一条用户消息
      for (let i = prompt.length - 1; i >= 0; i--) {
        if (prompt[i].role === 'user') {
          return typeof prompt[i].content === 'string' 
            ? prompt[i].content 
            : prompt[i].content?.[0]?.text || '';
        }
      }
    }
    return '';
  }
  
  private formatMemories(memories: Memory[]): string {
    const lines = memories.map((m, i) => {
      const date = new Date(m.createdAt).toLocaleDateString('zh-CN');
      return `${i + 1}. [${date}] ${m.content.task}`;
    });
    
    return `<relevant-memories>
以下是你之前与用户交流的相关记忆，供参考：
${lines.join('\n')}
</relevant-memories>`;
  }
}
```

### 1.8 AutoCapture (自动捕获)

#### 职责
- 在对话结束后强制生成记忆
- 分析并存储

```typescript
export class AutoCapture {
  private memorySystem: MemorySystem;
  private config: MemoryConfig;
  private triggerAnalyzer: TriggerAnalyzer;
  private categoryAnalyzer: CategoryAnalyzer;
  
  constructor(
    memorySystem: MemorySystem, 
    config: MemoryConfig
  ) {
    this.memorySystem = memorySystem;
    this.config = config;
    this.triggerAnalyzer = new TriggerAnalyzer();
    this.categoryAnalyzer = new CategoryAnalyzer();
  }
  
  async execute(event: any): Promise<void> {
    if (!this.config.autoCapture) return;
    
    try {
      // 1. 检查存储策略
      if (this.config.captureStrategy === 'selective') {
        // 关键词模式: 有关键词才存储
        const hasKeyword = this.checkKeywords(event.messages);
        if (!hasKeyword) return;
      }
      
      // 2. 生成对话摘要
      const summary = this.generateSummary(event.messages);
      if (!summary) return;
      
      // 3. 分析分类
      const { category, subCategory } = this.categoryAnalyzer.analyze(summary.task);
      
      // 4. 判断重要性
      const importance = this.triggerAnalyzer.analyze(summary.task);
      
      // 5. 提取标签
      const tags = this.triggerAnalyzer.extractTags(summary.task);
      
      // 6. 强制存储 (100%)
      await this.memorySystem.store({
        content: summary,
        category,
        subCategory,
        importance,
        tags
      });
      
      console.log(`[AutoCapture] 已存储记忆: ${category}/${subCategory}`);
      
    } catch (e) {
      console.error('自动捕获失败:', e);
    }
  }
  
  private checkKeywords(messages: any[]): boolean {
    const allText = messages
      .map(m => typeof m.content === 'string' ? m.content : m.content?.[0]?.text || '')
      .join(' ');
    
    const importance = this.triggerAnalyzer.analyze(allText);
    return importance !== 'low';
  }
  
  private generateSummary(messages: any[]): {
    task: string;
    process: string;
    result: string;
    insights: string[];
  } | null {
    // 提取用户消息和 AI 回复
    const userMessages: string[] = [];
    const assistantMessages: string[] = [];
    
    for (const msg of messages) {
      const content = typeof msg.content === 'string' 
        ? msg.content 
        : msg.content?.[0]?.text || '';
      
      if (msg.role === 'user') {
        userMessages.push(content);
      } else if (msg.role === 'assistant') {
        assistantMessages.push(content);
      }
    }
    
    if (userMessages.length === 0) return null;
    
    const lastUserMsg = userMessages[userMessages.length - 1];
    const lastAssistantMsg = assistantMessages[assistantMessages.length - 1] || '';
    
    // 限制长度
    const maxLen = this.config.captureMaxChars;
    
    return {
      task: lastUserMsg.substring(0, maxLen),
      process: '',
      result: lastAssistantMsg.substring(0, maxLen),
      insights: []
    };
  }
}
```
