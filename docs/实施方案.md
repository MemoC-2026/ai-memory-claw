#### 1.4 插件入口 index.ts

```typescript
import type { OpenClawPluginApi } from "openclaw/plugin-sdk";
import { memoryConfigSchema } from "./config.js";
import { MemorySystem } from "./memory-system.js";
import { AutoRecall } from "./auto-recall.js";
import { AutoCapture } from "./auto-capture.js";

let memorySystem: MemorySystem | null = null;

const plugin = {
  id: "ai-memory-claw",
  name: "AI Memory Claw",
  description: "AI渐进式记忆系统 - 无感运行版",
  kind: "memory" as const,
  configSchema,
  
  register(api: OpenClawPluginApi) {
    const config = memoryConfigSchema.parse(api.pluginConfig);
    const dataDir = api.resolvePath(config.dataDir || "~/.ai-memory-claw/data");
    
    // 初始化记忆系统
    memorySystem = new MemorySystem(dataDir, config);
    
    // 初始化自动召回
    const autoRecall = new AutoRecall(memorySystem, config);
    
    // 初始化自动捕获
    const autoCapture = new AutoCapture(memorySystem, config);
    
    // 注册 before_agent_start 钩子
    if (config.autoRecall) {
      api.on("before_agent_start", autoRecall.execute.bind(autoRecall));
    }
    
    // 注册 agent_end 钩子
    if (config.autoCapture) {
      api.on("agent_end", autoCapture.execute.bind(autoCapture));
    }
    
    // 注册可选工具
    api.registerTool({ ... }, { name: "ai_memory_search" });
    api.registerTool({ ... }, { name: "ai_memory_list" });
    api.registerTool({ ... }, { name: "ai_memory_stats" });
    
    api.logger.info("AI Memory Claw 插件已加载");
  }
};

export default plugin;
```

#### 3.2.3 triggers.ts

```typescript
// 关键词分析，用于判断重要性等级
export class TriggerAnalyzer {
  private criticalPatterns = [
    /记住|别忘了|提醒我/i,
    /密码|账号|身份|银行卡/i,
    /\+\d{10,}/,  // 电话
    /[\w.-]+@[\w.-]+\.\w+/  // 邮箱
  ];
  
  private highPatterns = [
    /决定|选择|采用|否决/i,
    /配置|安装|部署/i,
    /prefer|decide|choose/i
  ];
  
  private mediumPatterns = [
    /实现|修复|创建|帮助|完成/i,
    /implement|fix|create|help|complete/i
  ];
  
  analyze(content: string): ImportanceLevel {
    if (this.criticalPatterns.some(p => p.test(content))) {
      return 'critical';
    }
    if (this.highPatterns.some(p => p.test(content))) {
      return 'high';
    }
    if (this.mediumPatterns.some(p => p.test(content))) {
      return 'medium';
    }
    return 'low';
  }
}
```

#### 4.2.1 auto-recall.ts

```typescript
export class AutoRecall {
  private memorySystem: MemorySystem;
  private config: MemoryConfig;
  
  async execute(event: AgentStartEvent) {
    // 1. 获取用户输入
    const prompt = this.extractUserPrompt(event.prompt);
    if (!prompt || prompt.length < 3) return;
    
    // 2. 搜索相关记忆
    const memories = await this.memorySystem.search(prompt, {
      limit: this.config.recallLimit,
      threshold: this.config.recallThreshold
    });
    
    if (memories.length === 0) return;
    
    // 3. 格式化注入
    const context = this.formatMemories(memories);
    
    // 4. 返回注入上下文
    return {
      prependContext: context
    };
  }
  
  private formatMemories(memories: Memory[]): string {
    const lines = memories.map((m, i) => {
      return `${i + 1}. [${m.category}/${m.subCategory}] ${m.content.task}\n   结果: ${m.content.result}`;
    });
    
    return `<relevant-memories>
以下是你之前与用户交流的相关记忆，供参考：
${lines.join('\n')}
</relevant-memories>`;
  }
}
```

#### 4.2.2 auto-capture.ts

```typescript
export class AutoCapture {
  private memorySystem: MemorySystem;
  private config: MemoryConfig;
  private triggerAnalyzer: TriggerAnalyzer;
  
  async execute(event: AgentEndEvent) {
    // 1. 生成对话摘要 (强制 100%)
    const summary = this.generateSummary(event.messages);
    if (!summary) return;
    
    // 2. 分析话题分类
    const { category, subCategory } = this.analyzeCategory(summary);
    
    // 3. 判断重要性
    const importance = this.triggerAnalyzer.analyze(summary);
    
    // 4. 强制存储 (100% 执行)
    await this.memorySystem.store({
      content: {
        task: summary.task,
        process: summary.process,
        result: summary.result,
        insights: summary.insights
      },
      category,
      subCategory,
      importance,
      tags: summary.tags
    });
    
    // 5. 触发后台处理
    this.scheduleBackgroundTasks();
  }
  
  private generateSummary(messages: any[]): ConversationSummary {
    // 从消息中提取关键信息
    const userMessages = messages.filter(m => m.role === 'user');
    const assistantMessages = messages.filter(m => m.role === 'assistant');
    
    const lastUserMsg = userMessages[userMessages.length - 1]?.content || '';
    const lastAssistantMsg = assistantMessages[assistantMessages.length - 1]?.content || '';
    
    return {
      task: lastUserMsg.substring(0, 200),
      process: this.extractProcess(lastAssistantMsg),
      result: this.extractResult(lastAssistantMsg),
      insights: this.extractInsights(lastAssistantMsg),
      tags: this.extractTags(lastUserMsg + lastAssistantMsg)
    };
  }
}
```

#### 4.2.3 forgetter.ts

```typescript
export class Forgrtter {
  async checkAndForget(memories: Memory[]): Promise<ForgetResult[]> {
    const results: ForgetResult[] = [];
    const now = Date.now();
    
    for (const memory of memories) {
      const daysSinceAccess = (now - new Date(memory.lastAccessedAt).getTime()) / DAY_MS;
      const importanceScore = this.getImportanceScore(memory.importance);
      
      let shouldForget = false;
      
      // 遗忘条件
      if (daysSinceAccess > 30 && importanceScore < 0.3) {
        shouldForget = true;
      } else if (daysSinceAccess > 60 && importanceScore < 0.5) {
        shouldForget = true;
      } else if (daysSinceAccess > 90) {
        shouldForget = true;
      }
      
      if (shouldForget) {
        if (importanceScore > 0) {
          // 先压缩
          await this.compress(memory);
          results.push({ id: memory.id, action: 'compressed' });
        } else {
          // 再删除
          await this.delete(memory.id);
          results.push({ id: memory.id, action: 'deleted' });
        }
      }
    }
    
    return results;
  }
}
```

#### 4.2.4 integrator.ts

```typescript
export class Integrator {
  async integrate(memories: Memory[]): Promise<IntegrationResult[]> {
    const results: IntegrationResult[] = [];
    
    // 1. 发现相似的记忆群
    const clusters = await this.discoverClusters(memories);
    
    for (const cluster of clusters) {
      if (cluster.memories.length < 2) continue;
      
      // 2. 生成簇摘要
      const summary = await this.generateClusterSummary(cluster);
      
      // 3. 决定是否合并
      if (this.shouldMerge(cluster)) {
        await this.mergeIntoCluster(cluster, summary);
        results.push({ 
          clusterId: cluster.id, 
          action: 'merged',
          newMemoryId: summary.id 
        });
      } else {
        await this.updateClusterSummary(cluster.id, summary);
        results.push({ 
          clusterId: cluster.id, 
          action: 'updated' 
        });
      }
    }
    
    return results;
  }
}
```
